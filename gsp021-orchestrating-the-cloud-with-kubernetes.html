Pod
pod内容器共享同一网络和存储
创建Pod
# monolith.yaml apiVersion: v1 kind: Pod metadata: name: monolith labels: app: monolith spec: containers: - name: monolith image: kelseyhightower/monolith:1.0.0 args: - "-http=0.0.0.0:80" - "-health=0.0.0.0:81" - "-secret=secret" ports: - name: http containerPort: 80 - name: health containerPort: 81 resources: limits: cpu: 0.2 memory: "10Mi"
Bash
kubectl create -f pods/monolith.yaml kubectl get pods kubectl describe pods monolith curl http://127.0.0.1:10080

与Pod交互
# 本地映射端口 kubectl port-forward monolith 10080:80 # 容器自身服务的内容，与Pod无关 curl http://127.0.0.1:10080/secure TOKEN=$(curl http://127.0.0.1:10080/login -u user|jq -r '.token') curl -H "Authorization: Bearer $TOKEN" http://127.0.0.1:10080/secure # 查看日志 kubectl logs monolith curl http://127.0.0.1:10080 # 与Pod交互 kubectl exec monolith --stdin --tty -c monolith /bin/sh

# 通过secret创建证书 kubectl create secret generic tls-certs --from -file tls/ # 通过comfigmap创建配置文件 kubectl create configmap nginx-proxy-conf --from-file nginx/proxy.conf kubectl create -f pods/secure-monolith.yaml
Bash
使用标签选择器向外部公开一组有限的 Pod
gcloud compute firewall-rules create allow-monolith-nodeport --allow=tcp:31000 gcloud compute instances list curl -k https://<ip>:31000
Bash
添加标签
由于service是通过标签来寻找匹配的Pod，所以这里需要为Pod添加对应的标签
kubectl get pods -l "app=monolith" # 查看特定标签的Pod kubectl get pods -l "app=monolith,secure=enabled" # 添加标签 kubectl label pods secure-monolith 'secure=enabled' kubectl get pods secure-monolith --show-labels # services匹配到Pod kubectl describe services monolith | grep Endpoints
